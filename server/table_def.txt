CREATE TABLE `deliveries` (
  `id` int NOT NULL AUTO_INCREMENT,
  `uuid` varchar(36) NOT NULL,
  `business_id` int NOT NULL,
  `driver_id` int DEFAULT NULL,
  `customer_id` int NOT NULL,
  `order_number` varchar(100) DEFAULT NULL,
  `delivery_status` enum('pending','dispatched','en_route','arrived','delivered','failed','disputed','cancelled') DEFAULT 'pending',
  `scheduled_time` timestamp NULL DEFAULT NULL,
  `estimated_arrival` timestamp NULL DEFAULT NULL,
  `actual_arrival` timestamp NULL DEFAULT NULL,
  `actual_departure` timestamp NULL DEFAULT NULL,
  `delivery_notes` text,
  `requires_signature` tinyint(1) DEFAULT '1',
  `requires_photo` tinyint(1) DEFAULT '1',
  `priority_level` enum('low','medium','high') DEFAULT 'medium',
  `route_optimization_data` json DEFAULT NULL,
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `photo_url` text,
  `signature_url` text,
  `gps_lat` decimal(10,8) DEFAULT NULL,
  `gps_lng` decimal(11,8) DEFAULT NULL,
  `gps_accuracy` float DEFAULT NULL,
  `proof_hash` varchar(255) DEFAULT NULL,
  `recorded_at` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uuid` (`uuid`),
  KEY `idx_delivery_uuid` (`uuid`),
  KEY `idx_delivery_status` (`delivery_status`),
  KEY `idx_delivery_business` (`business_id`,`delivery_status`),
  KEY `idx_delivery_driver` (`driver_id`,`scheduled_time`),
  KEY `idx_delivery_customer` (`customer_id`,`created_at`),
  KEY `idx_delivery_scheduled` (`scheduled_time`),
  CONSTRAINT `deliveries_ibfk_1` FOREIGN KEY (`business_id`) REFERENCES `businesses` (`id`) ON DELETE CASCADE,
  CONSTRAINT `deliveries_ibfk_2` FOREIGN KEY (`driver_id`) REFERENCES `drivers` (`id`) ON DELETE SET NULL,
  CONSTRAINT `deliveries_ibfk_3` FOREIGN KEY (`customer_id`) REFERENCES `customers` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=26 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
